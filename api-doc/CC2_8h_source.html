184  bool
185  /// Iterates the singles equations for CCS, CC2, LRCC2
186  /// The corresponding regulairzation tails of the doubles are updated in every iteration (therefore doubles are not marked as const)
187  /// @param[in] : singles, the singles that are iterated
188  /// @param[in] : singles2, additional set of singles for potential (in LRCC2 this are the Ground State singles)
189  /// @param[in] : gs_doubles, Ground State doubles (Needed for CC2 and LRCC2)
190  /// @param[in] : ex_doubles, Excited State doubles (Needed for LRCC2)
191  /// @param[in] : ctype: the calculation type: CCS, CC2, CC2_response_
192  /// @param[in] : maxiter: maxmial number of iterations
193  /// @param[out]: true if the overall change of the singles is below 10*donv_6D
194  iterate_singles(CC_vecfunction& singles, const CC_vecfunction singles2, Pairs<CCPair>& gs_doubles,
195  Pairs<CCPair>& ex_doubles, const CalcType ctype, const std::size_t maxiter) {
196  output.subsection("Iterate " + assign_name(ctype) + "-Singles");
197  CCTimer time_all(world, "Overall Iteration of " + assign_name(ctype) + "-Singles");
198  bool converged = true;
199 
200  CC_vecfunction old_singles(singles);
201  for (auto& tmp : singles.functions)
202  old_singles(tmp.first).function = copy(tmp.second.function);
203 
204  // KAIN solver
205  typedef vector_function_allocator<double, 3> allocT;
206  typedef XNonlinearSolver<std::vector<Function<double, 3> >, double, allocT> solverT;
207  allocT alloc(world, singles.size());
208  solverT solver(allocT(world, singles.size()));
209  solver.do_print = (world.rank() == 0);
210 
211  for (size_t iter = 0; iter < maxiter; iter++) {
212  output.subsection("Microiteration " + std::to_string(iter) + " of " + assign_name(ctype) + "-Singles");
213  CCTimer time(world, "Microiteration " + std::to_string(iter) + " of " + assign_name(ctype) + "-Singles");
214  double omega = 0.0;
215  if (ctype == CT_LRCC2) omega = singles.omega;
216  else if (ctype == CT_LRCCS) omega = singles.omega;
217  else if (ctype == CT_ADC2) omega = singles.omega;
218 
219  // consistency check
220  switch (ctype) {
221  case CT_CC2:
222  if (singles.type != PARTICLE)
223  output.warning("iterate_singles: CC2 demanded but singles are not of type PARTICLE");
224  break;
225  case CT_MP2: MADNESS_EXCEPTION("Demanded Singles Calculation for MP2 ????", 1);
226  break;
227  case CT_LRCC2:
228  if (singles.type != RESPONSE or singles2.type != PARTICLE)
229  output.warning("iterate_singles: CC2_response_ singles have wrong types");
230  break;
231  case CT_LRCCS:
232  if (singles.type != RESPONSE)
233  output.warning("iterate_singles: CCS_response_ singles have wrong types");
234  break;
235  case CT_CISPD: MADNESS_EXCEPTION("Demanded Singles Calculation for CIS(D)", 1);
236  break;
237  case CT_ADC2:
238  MADNESS_ASSERT(singles.type == RESPONSE);
239  break;
240  case CT_TEST: MADNESS_EXCEPTION("Iterate Singles not implemented for Experimental calculation", 1);
241  break;
242  default: MADNESS_EXCEPTION(
243  ("Unknown calculation type in iterate singles: " + assign_name(ctype)).c_str(), 1);
244  }
245 
246  // get potentials
247  CCTimer time_V(world, assign_name(ctype) + "-Singles Potential");
248  vector_real_function_3d V;
249  if (ctype == CT_CC2) V = CCOPS.get_CC2_singles_potential_gs(singles, gs_doubles);
250  else if (ctype == CT_LRCC2)
251  V = CCOPS.get_CC2_singles_potential_ex(singles2, gs_doubles, singles, ex_doubles);
252  else if (ctype == CT_LRCCS) V = CCOPS.get_CCS_potential_ex(singles);
253  else if (ctype == CT_ADC2) V = CCOPS.get_ADC2_singles_potential(gs_doubles, singles, ex_doubles);
254  else MADNESS_EXCEPTION("iterate singles: unknown type", 1);
255  time_V.info(true, norm2(world, V));
256 
257  if (ctype == CT_LRCCS or ctype == CT_LRCC2 or ctype == CT_ADC2) {
258  omega = singles.omega; // computed with the potential
259  }
260 
261  scale(world, V, -2.0);
262  truncate(world, V);
263 
264  // make bsh operators
265  CCTimer time_makebsh(world, "Make G-Operators");
266  std::vector<std::shared_ptr<SeparatedConvolution<double, 3> > > G(singles.size());
267  for (size_t i = 0; i < G.size(); i++) {
268  const double bsh_eps = CCOPS.get_orbital_energies()[i + parameters.freeze()] + omega;
269  G[i] = std::shared_ptr<SeparatedConvolution<double, 3> >(
270  BSHOperatorPtr3D(world, sqrt(-2.0 * bsh_eps), parameters.lo(), parameters.thresh_bsh_3D()));
271  }
272  world.gop.fence();
273  time_makebsh.info();
274 
275  // apply bsh operators
276  CCTimer time_applyG(world, "Apply G-Operators");
277  vector_real_function_3d GV = apply<SeparatedConvolution<double, 3>, double, 3>(world, G, V);
278  world.gop.fence();
279  time_applyG.info();
280 
281  // apply Q-Projector to result
282  GV = CCOPS.apply_Qt(GV, CCOPS.mo_ket());
283 
284  // Normalize Singles if it is excited state
285  if (ctype == CT_LRCCS or ctype == CT_LRCC2 or ctype == CT_ADC2) {
286  output("Normalizing new singles");
287  const vector_real_function_3d x = GV;
288  const vector_real_function_3d xbra = mul(world, nemo->ncf->square(), GV);
289  const double norm = sqrt(inner(world, xbra, x).sum());
290  if (world.rank() == 0)
291  std::cout << " Norm was " << std::fixed << std::setprecision(parameters.output_prec()) << norm
292  << "\n";
293  scale(world, GV, 1.0 / norm);
294  } else output("Singles not normalized");
295 
296  // residual
297  const vector_real_function_3d residual = sub(world, singles.get_vecfunction(), GV);
298 
299  // information
300  const Tensor<double> R2xinnerx = inner(world, mul(world, nemo->ncf->square(), singles.get_vecfunction()),
301  singles.get_vecfunction());
302  const Tensor<double> R2GVinnerGV = inner(world, mul(world, nemo->ncf->square(), GV), GV);
303  const Tensor<double> R2rinnerr = inner(world, mul(world, nemo->ncf->square(), residual), residual);
304  const double R2vector_error = sqrt(R2rinnerr.sum());
305 
306  // print information
307  if (world.rank() == 0) std::cout << "\n\n-----Results of current interation:-----\n";
308  if (world.rank() == 0)
309  std::cout << "\nName: ||" << singles.name(0) << "||, ||GV" << singles.name(0) << ", ||residual||" << "\n";
310  if (world.rank() == 0)
311  std::cout << singles.name(0) << ": " << std::scientific << std::setprecision(parameters.output_prec())
312  << sqrt(R2xinnerx.sum()) << ", " << sqrt(R2GVinnerGV.sum()) << ", " << sqrt(R2rinnerr.sum())
313  << "\n----------------------------------------\n";
314  for (size_t i = 0; i < GV.size(); i++) {
315  if (world.rank() == 0)
316  std::cout << singles(i + parameters.freeze()).name() << ": " << std::scientific
317  << std::setprecision(parameters.output_prec())
318  << sqrt(R2xinnerx(i)) << ", " << sqrt(R2GVinnerGV(i)) << ", " << sqrt(R2rinnerr(i))
319  << "\n";
320  }
321  if (world.rank() == 0) std::cout << "\n----------------------------------------\n\n";
322 
323  // make second order update (only for response)
324  if (ctype == CT_LRCC2 or ctype == CT_LRCCS) {
325  output("\nMake 2nd order energy update:");
326  // include nuclear factors
327  {
328  vector_real_function_3d bra_res = mul(world, nemo->ncf->square(), residual);
329  vector_real_function_3d bra_GV = mul(world, nemo->ncf->square(), GV);
330  double Rtmp = inner(world, bra_res, V).sum();
331  double Rtmp2 = inner(world, bra_GV, GV).sum();
332  const double Rdelta = (0.5 * Rtmp / Rtmp2);
333  double old_omega = omega;
334  output("Delta-Update is not used");
335  if (world.rank() == 0)
336  std::cout << "omega, old_omega, delta" << std::fixed
337  << std::setprecision(parameters.output_prec() + 2) << omega << ", " << old_omega << ", "
338  << Rdelta << "\n\n";
339  }
340 
341  }
342 
343  // update singles
344  singles.omega = omega;
345  vector_real_function_3d new_singles = GV;
346  if (parameters.kain()) new_singles = solver.update(singles.get_vecfunction(), residual);
347  print_size(world, new_singles, "new_singles");
348  truncate(world, new_singles);
349  print_size(world, new_singles, "new_singles");
350  for (size_t i = 0; i < GV.size(); i++) {
351  singles(i + parameters.freeze()).function = copy(new_singles[i]);
352  }
353 
354  // update intermediates
355  CCOPS.update_intermediates(singles);
356 
357  // update reg_residues of doubles
358  //if(ctype==CC2_) update_reg_residues_gs(singles,gs_doubles);
359  //else if(ctype==LRCC2_) update_reg_residues_ex(singles2,singles,ex_doubles);
360 
361  converged = (R2vector_error < parameters.dconv_3D());
362 
363  time.info();
364  if (converged) break;
365  if (ctype == CT_LRCCS) break; // for CCS just one iteration to check convergence
366  }
367  time_all.info();
368 
369  // Assign the overall changes
370  bool no_change = true;
371  if (world.rank() == 0)
372  std::cout << "Change in Singles functions after all the CC2-Single-Microiterations" << std::endl;
373  for (auto& tmp : singles.functions) {
374  const double change = (tmp.second.function - old_singles(tmp.first).function).norm2();
375  tmp.second.current_error = change;
376  if (change > parameters.dconv_3D()) no_change = false;
377  if (world.rank() == 0)
378  std::cout << "Change of " << tmp.second.name() << "=" << tmp.second.current_error << std::endl;
379  }
380  // update reg_residues of doubles
381  if (ctype == CT_CC2) update_reg_residues_gs(singles, gs_doubles);
382  else if (ctype == CT_LRCC2) update_reg_residues_ex(singles2, singles, ex_doubles);
383 
384  //CCOPS.plot(singles);
385  if (no_change) output("Change of Singles was below = " + std::to_string(parameters.dconv_3D()) + "!");
386  return no_change;
387  }
