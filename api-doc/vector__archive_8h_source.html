68  /// Appends data to the end of the vector.
69 
70  /// \todo Verify/complete the documentation.
71  /// \tparam T The type of data to be appended.
72  /// \param[in] t Pointer to the data to be appended.
73  /// \param[in] n The number of data items to be appended.
74  /// \return Description needed.
75  template <class T>
76  inline
77  typename std::enable_if< madness::is_trivially_serializable<T>::value, void >::type
78  store(const T* t, long n) const {
79  const unsigned char* ptr = (unsigned char*) t;
80  v->insert(v->end(),ptr,ptr+n*sizeof(T));

99  /// Wraps an archive around an STL \c vector for input.
100  class VectorInputArchive : public BaseInputArchive {
101  mutable std::vector<unsigned char>* v; ///< The STL vector being wrapped.
102  mutable std::size_t i; ///< Current input location.
103 
104  public:
105  /// Create a buffer to wrap the specified \c vector.
106 
107  /// \param[in] v The \c vector.
108  VectorInputArchive(std::vector<unsigned char>& v) : v(&v) , i(0) {}
109 
110  /// Load data from the vector.
111 
112  /// The function only appears (due to \c enable_if) if \c T is
113  /// serializable.
114  /// \tparam T The type of data to be loaded.
115  /// \param[out] t Where to store the loaded data.
116  /// \param[in] n The number of data items to be loaded.
117  template <class T>
118  inline
119  typename std::enable_if< madness::is_trivially_serializable<T>::value, void >::type
120  load(T* t, long n) const {
121  std::size_t m = n*sizeof(T);
122  if (m+i > v->size()) MADNESS_EXCEPTION("VectorInputArchive: reading past end", m+1);
123  memcpy((unsigned char*) t, &((*v)[i]), m);
124  i += m;
125  }
126 
127  /// Open the archive.
128  void open() {};
129 
130  /// Reset the read location to the beginning of the \c vector.
131  void rewind() const {
132  i=0;
133  };
134 
135  /// Get the amount of space left to be read from the \c vector.
136 
137  /// \return The amount of space left to be read from the \c vector.
138  std::size_t nbyte_avail() const {
139  return v->size()-i;
140  };
141 
142  /// Close the archive.
143  void close() {}
144  };
