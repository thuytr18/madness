17  /// index in packed lower triangle matrix, row >= col
18  template <typename Int>
19  size_t tridx(Int row, Int col) {
20  return (row < col) ? (row + col * (col + 1) / 2)
21  : (col + row * (row + 1) / 2);
22  }
