110  /// Invoking this switches all registered objects from this process map to the new one
111 
112  /// After invoking this routine all objects will be registered with the
113  /// new map and no objects will be registered in the current map.
114  /// @param[in] world The associated world
115  /// @param[in] newpmap The new process map
116  void redistribute(World& world, const std::shared_ptr< WorldDCPmapInterface<keyT> >& newpmap) {
117  print_data_sizes(world, "before redistributing");
118  world.gop.fence();
119  for (typename std::set<ptrT>::iterator iter = ptrs.begin();
120  iter != ptrs.end();
121  ++iter) {
122  (*iter)->redistribute_phase1(newpmap);
123  }
124  world.gop.fence();
125  for (typename std::set<ptrT>::iterator iter = ptrs.begin();
126  iter != ptrs.end();
127  ++iter) {
128  (*iter)->redistribute_phase2();
129  newpmap->register_callback(*iter);
130  }
131  world.gop.fence();
132  for (typename std::set<ptrT>::iterator iter = ptrs.begin();
133  iter != ptrs.end();
134  ++iter) {
135  (*iter)->redistribute_phase3();
136  }
137  world.gop.fence();
138  ptrs.clear();
139  newpmap->print_data_sizes(world, "after redistributing");
140  }

142  /// Counts global number of entries in all containers associated with this process map
143 
144  /// Collective operation with global fence
145  std::size_t global_size(World& world) const {
146  world.gop.fence();
147  std::size_t sum = local_size();
148  world.gop.sum(sum);

167  std::size_t total = global_size(world);
168  std::vector<std::size_t> sizes(world.size());

170  world.gop.sum(&sizes[0],world.size());
171  if (world.rank() == 0) {
172  madness::print("data distribution info", msg);
173  madness::print(" total: ", total);
174  std::cout << " procs: ";
175  for (int i=0; i<world.size(); i++) std::cout << sizes[i] << " ";

178  world.gop.fence();

753  /// There is no communication or syncronization associated with

1618  /// \ingroup worlddc
1619  /// Each node (process) is served by a designated IO node.
1620  /// The IO node has a binary local file archive to which is
1621  /// first written a cookie and the number of servers. The IO
1622  /// node then loops thru all of its clients and in turn tells
1623  /// each to write its data over an MPI stream, which is copied
1624  /// directly to the output file. The stream contents are then
1625  /// cookie, no. of clients, foreach client (usual sequential archive).
1626  ///
1627  /// If ar.dofence() is true (default) fence is invoked before and
1628  /// after the IO. The fence is optional but it is of course
1629  /// necessary to be sure that all updates have completed
1630  /// before doing IO, and that all IO has completed before
1631  /// subsequent modifications. Also, there is always at least
1632  /// some synchronization between a client and its IO server.
1633  template <class keyT, class valueT, class localarchiveT>
1634  struct ArchiveStoreImpl< ParallelOutputArchive<localarchiveT>, WorldContainer<keyT,valueT> > {
