80  /// project f on p:
81 
82  /// \f[
83  /// | result > = \sum_p | p > <p|f>
84  /// \f]
85  /// @param[in] f the function to be projected
86  /// @return the projection of f on the space of p
87  funcT operator()(const funcT& f) const {
88  return this->operator()(vecfuncT(1,f)).front();
89  }

91  /// project f on p:
92 
93  /// \f[
94  /// | result > = \sum_p | p > <p|f>
95  /// \f]
96  /// @param[in] f the vector of functions to be projected
97  /// @return the projection of f on the space of p
98  vecfuncT operator()(const vecfuncT& f) const {
99  if (f.size()==0) return vecfuncT();
100  World& world=f[0].world();
101  Tensor<T> ovlp=matrix_inner(world,mo_bra_,f);
102  vecfuncT result=transform(world,mo_ket_,ovlp,true);
103  truncate(world,result);
104  return result;
105  }

107  /// apply 3D Projector to one particle of a 6D function
108  /// \f[
109  /// |result> = \sum_p |p(particle)> <p(particle)|f(1,2)>_{particle}
110  /// \f]
111  /// @param[in] f the 6D function to be projected
112  /// @param[in] the particle that is projected (1 or 2)
113  /// @return the projected function
114  template<std::size_t KDIM>
115  typename std::enable_if<KDIM==2*NDIM, Function<T,KDIM> >::type
116  operator()(const Function<T,KDIM>& f, size_t particle1=size_t(-1)) const {
117  Function<T,KDIM> result = FunctionFactory<T,KDIM>(f.world());
118  if (particle1==size_t(-1)) particle1=particle;
119  MADNESS_CHECK_THROW(particle1 == 1 or particle1 == 2, "particle must be 1 or 2");
120  for (size_t i = 0; i < mo_ket_.size(); i++) {
121  Function<T,NDIM> tmp1 = mo_ket_[i];
122  Function<T,NDIM> tmp2 = f.project_out(mo_bra_[i], particle1 - 1);
123  Function<T,KDIM> tmp12;
124  if (particle1 == 1) {
125  tmp12 = CompositeFactory<T, KDIM, NDIM>(f.world()).particle1(copy(tmp1)).particle2(copy(tmp2));
126  tmp12.fill_tree();
127  } else {
128  tmp12 = CompositeFactory<T, KDIM, NDIM>(f.world()).particle1(copy(tmp2)).particle2(copy(tmp1));
129  tmp12.fill_tree();
130  }
131  result += tmp12;
132  }
133  return result;
134  }

275  /// apply the projection parts of the strong orthogonality operator Q12 on a function f
276 
277  /// The SO operator is defined as 1-O1-O2+O1O2, where O1 and O2 are projectors
278  /// return the term -O1-O2+O1O2 only, such that Q12 f = 1 + outer(result.first,result.second)
279  std::pair<std::vector<Function<T,NDIM>>,std::vector<Function<T,NDIM>>>
280  get_vectors_for_outer_product(const Function<T,2*NDIM>& f) const {
281  // Eq. (A9): g_kl = < k(1) l(2) | f(1,2) >
282  // note no (kl) symmetry here!
283  reconstruct(world, bra1_, false);
284  reconstruct(world, bra2_, true);
285  Tensor<double> g_kl(bra1_.size(), bra2_.size());
286  for (size_t k = 0; k < bra1_.size(); ++k) {
287  for (size_t l = 0; l < bra2_.size(); ++l) {
288  Function<T, 2 * NDIM> kl = CompositeFactory<T, 2 * NDIM, NDIM>(world)
289  .particle1(bra1_[k]).particle2(bra2_[l]);
290  g_kl(k, l) = inner(f, kl);
291  }
292  }
293 
294  // Eq. (A12)
295  // project out the mainly first particle: O1 (1 - 1/2 O2)
296  std::vector<Function<T, NDIM>> h2(bra1_.size());
297  std::vector<Function<T, NDIM>> h1(ket2_.size());
298  reconstruct(world, bra1_, false);
299  reconstruct(world, bra2_, true);
300  for (size_t k = 0; k < bra1_.size(); ++k) {
301 
302  // project out the mainly first particle: O1 (1 - 1/2 O2): first term
303  // Eq. (A10)
304  h2[k] = f.project_out(bra1_[k], 0);
305 
306  // project out the mainly second particle: O2 (1 - 1/2 O1): first term
307  // Eq. (A11)
308  std::size_t l = k;
309  h1[l] = f.project_out(bra2_[l], 1);
310 
311  }
312 
313  // Transforms a vector of functions according to new[i] = sum[j] old[j]*c[j,i]
314  // project out the mainly first particle: O1 (1 - 1/2 O2): second term
315  // Eq. (A12), (A13)
316  h2 -= transform(world, ket2_, 0.5 * transpose(g_kl), false); // ordering g(k,l) is correct
317  h1 -= transform(world, ket1_, 0.5 * g_kl, true); // ordering g(k,l) is correct
318  // aka
319  // for (size_t l=0; l<ket2_.size(); ++l) {
320  // h2[k]-=0.5*g_kl(k,l)*ket2_[l];
321  // }
322 
323  change_tree_state(h1, reconstructed, false);
324  change_tree_state(h2, reconstructed, false);
325  change_tree_state(ket1_, reconstructed, false);
326  change_tree_state(ket2_, reconstructed, false);
327  world.gop.fence();
328 
329  auto left=append(ket1_,h1);
330  auto right=append(h2,ket2_);
331  return std::make_pair(-1.0*left,right);
332  }

334  /// apply the strong orthogonality operator Q12 on a function f
335 
336  /// notation of the equations follows
337  /// J. Chem. Phys., vol. 139, no. 11, p. 114106, 2013.
338  Function<T,2*NDIM> operator()(const Function<T,2*NDIM>& f) const {
339 
340  // simple and it works for higher accuracies, but might be
341  // imprecise for lower accuracies
342 // return (f-O1(f)-O2(f)+O1(O2(f))).truncate().reduce_rank();
343 
344  auto [left,right]=get_vectors_for_outer_product(f);
345 
346  // temporarily tighten the threshold
347  double thresh=FunctionDefaults<2*NDIM>::get_thresh();
348  double tight_thresh=thresh*0.1;
349  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
350 
351  auto tmp=hartree_product(left,right);
352  tmp.truncate(thresh*0.3);
353 
354  FunctionDefaults<2*NDIM>::set_thresh(thresh);
355  Function<T, 2 * NDIM> result = copy(f);
356  result+=tmp;
357  result.truncate(thresh).reduce_rank();
358  return result;
359  }
