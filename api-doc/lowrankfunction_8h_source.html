129  std::vector<Vector<double,NDIM>> get_grid() const {
130  std::vector<Vector<double, NDIM>> grid;
131  long npoint_within_volume=volume()/volume_element;
132  if (do_print) print("npoint_within_volume",npoint_within_volume);
133 
134  auto cell = FunctionDefaults<NDIM>::get_cell();
135  auto is_in_cell = [&cell](const Vector<double, NDIM>& r) {
136  for (int d = 0; d < NDIM; ++d) if (r[d] < cell(d, 0) or r[d] > cell(d, 1)) return false;
137  return true;
138  };
139  double rad=radius;
140  auto o=origin;
141  auto is_in_sphere = [&rad,&o](const Vector<double, NDIM>& r) {
142  return ((r-o).normf()<rad);
143  };
144 
145  // set variance such that about 70% of all grid points sits within the radius
146  double variance=radius;
147  if (NDIM==2) variance=0.6*radius;
148  if (NDIM==3) variance=0.5*radius;
149  long maxrank=10*npoint_within_volume;
150  long rank=0;
151  for (int r=0; r<maxrank; ++r) {
152  auto tmp = gaussian_random_distribution(origin, variance);
153  if (not is_in_cell(tmp)) continue;
154  if (is_in_sphere(tmp)) ++rank;
155  grid.push_back(tmp);
156  if (rank==npoint_within_volume) break;
157  }
158  if (do_print) {
159  print("origin ",origin);
160  print("radius ",radius);
161  print("grid points in volume ",rank);
162  print("total grid points ",grid.size());
163  print("ratio ",rank/double(grid.size()));
164  print("volume element ",volume()/rank);
165  }
166  return grid;
167  }

466  std::vector<Function<T,LDIM>> result;
467  // functor is now \sum_i a_i(1) b_i(2) f12
468  // result(1) = \sum_i \int a_i(1) f(1,2) b_i(2) rhs(2) d2

475  auto begin= rhs.begin()+i;
476  auto end= (i+nbatch)<rhs.size() ? rhs.begin()+i+nbatch : rhs.end();

481

484  auto postmultiply= p1.is_first() ? b[ia] : a[ia];

488  auto tmp1=apply(world,*(f12),tmp);
489  if (postmultiply.is_initialized()) tmp1=tmp1*postmultiply;

491  }

775  /// after external operations g might not be orthonormal and/or optimal -- reorthonormalize
776 
777  /// orthonormalization similar to Bischoff, Harrison, Valeev, JCP 137 104103 (2012), Sec II C 3
778  /// f =\sum_i g_i h_i
779  /// = g X- (X+)^T (Y+)^T Y- h
780  /// = g X- U S V^T Y- h
781  /// = g (X- U) (S V^T Y-) h
782  /// requires 2 matrix_inner and 2 transforms. g and h are optimal, but contain all cusps etc..
783  /// @param[in] thresh SVD threshold
784  void reorthonormalize(double thresh=-1.0) {
785  if (thresh<0.0) thresh=rank_revealing_tol;
786  Tensor<T> ovlp_g = matrix_inner(world, g, g);
787  Tensor<T> ovlp_h = matrix_inner(world, h, h);
788  auto [eval_g, evec_g] = syev(ovlp_g);
789  auto [eval_h, evec_h] = syev(ovlp_h);
790 
791  // get relevant part of the eigenvalues and eigenvectors
792  // eigenvalues are sorted in ascending order
793  auto get_slice = [](auto eval, double thresh) {
794  // remove small/negative eigenvalues
795  eval.screen(thresh);
796  Slice s;
797  for (int i=0; i<eval.size(); ++i) {
798  MADNESS_CHECK_THROW(eval[i]>=0.0,"negative eigenvalues in reorthonormalize");
799  if (eval[i]>thresh) {
800  return s=Slice(i,-1); // from i to the end
801  break;
802  }
803  }
804  return s;
805  };
806 
807  Slice gslice=get_slice(eval_g,1.e-13);
808  Slice hslice=get_slice(eval_h,1.e-13);
809 
810  Tensor<T> Xplus=copy(evec_g(_,gslice));
811  Tensor<T> Xminus=copy(evec_g(_,gslice));
812  Tensor<T> Yplus=copy(evec_h(_,hslice));
813  Tensor<T> Yminus=copy(evec_h(_,hslice));
814  eval_g=copy(eval_g(gslice));
815  eval_h=copy(eval_h(hslice));
816 
817  for (int i=0; i<eval_g.size(); ++i) Xplus(_,i)*=std::pow(eval_g(i),0.5);
818  for (int i=0; i<eval_g.size(); ++i) Xminus(_,i)*=std::pow(eval_g(i),-0.5);
819  for (int i=0; i<eval_h.size(); ++i) Yplus(_,i)*=std::pow(eval_h(i),0.5);
820  for (int i=0; i<eval_h.size(); ++i) Yminus(_,i)*=std::pow(eval_h(i),-0.5);
821 
822  Tensor<T> M=madness::inner(Xplus,Yplus,0,0); // (X+)^T Y+
823  auto [U,s,VT]=svd(M);
824 
825  // truncate
826  typename Tensor<T>::scalar_type s_accumulated=0.0;
827  int i=s.size()-1;
828  for (;i>=0; i--) {
829  s_accumulated+=s[i];
830  if (s_accumulated>thresh) {
831  i++;
832  break;
833  }
834  }
835  for (int j=0; j<s.size(); ++j) VT(j,_)*=s[j];
836  Tensor<T> XX=madness::inner(Xminus,U,1,0);
837  Tensor<T> YY=madness::inner(Yminus,VT,1,1);
838 
839  g=truncate(transform(world,g,XX));
840  h=truncate(transform(world,h,YY));
841  }

862  /// compute the l2 error |functor - \sum_i g_ih_i|_2
863 
864  /// \int (f(1,2) - gh(1,2))^2 = \int f(1,2)^2 - 2\int f(1,2) gh(1,2) + \int gh(1,2)^2
865  /// since we are subtracting large numbers the numerics are sensitive, and NaN may be returned..
866  double l2error(const LRFunctorBase<T,NDIM>& lrfunctor1) const {
867 
868  timer t(world);
869  t.do_print=do_print;
870 
871  // \int f(1,2)^2 d1d2
872  double term1 = lrfunctor1.norm2();
873  term1=term1*term1;
874  t.tag("computing term1");
875 
876  // \int f(1,2) pre(1) post(2) \sum_i g(1) h(2) d1d2
877 // double term2=madness::inner(pre*g,f12(post*h));
878  double term2=madness::inner(g,inner(lrfunctor1,h,p2,p1));
879  t.tag("computing term2");
880 
881  // g functions are orthonormal
882  // \int gh(1,2)^2 d1d2 = \int \sum_{ij} g_i(1) g_j(1) h_i(2) h_j(2) d1d2
883  // = \sum_{ij} \int g_i(1) g_j(1) d1 \int h_i(2) h_j(2) d2
884  // = \sum_{ij} delta_{ij} \int h_i(2) h_j(2) d2
885  // = \sum_{i} \int h_i(2) h_i(2) d2
886  double zero=check_orthonormality(g);
887  if (zero>1.e-10) print("g is not orthonormal",zero);
888  // double term3a=madness::inner(h,h);
889  auto tmp1=matrix_inner(world,h,h);
890  auto tmp2=matrix_inner(world,g,g);
891  double term3=tmp1.trace(tmp2);
892 // print("term3/a/diff",term3a,term3,term3-term3a);
893  t.tag("computing term3");
894 
895  double arg=term1-2.0*term2+term3;
896  if (arg<0.0) {
897  print("negative l2 error");
898  arg*=-1.0;
899 // throw std::runtime_error("negative argument in l2error");
900  }
901  double error=sqrt(arg)/sqrt(term1);
902  if (world.rank()==0 and do_print) {
903  print("term1,2,3, error",term1, term2, term3, " --",error);
904  }
905 
906  return error;
907  }
