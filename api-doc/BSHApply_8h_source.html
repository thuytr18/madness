19 /// - constructs the bsh operator with the appropriate exponents
20 /// - performs a level shift if necessary
21 /// - adds coupling terms: ( T - fock(i,i) ) psi_i = -V psi_i + \sum_{j\neq i} psi_j fock(j,i)

43  /// apply the BSH operator on the vector of functions and respective potentials
44 
45  /// @param[in] eps orbital energies or the square fock matrix
46  /// @param[in] Vpsi vector of functions V*MOs
47  /// @return an MO structure holding the residuals and the orbital energy updates
48  std::tuple<std::vector<Function<T,NDIM> >, Tensor<double> > operator()(
49  const std::vector<Function<T,NDIM> > psi,
50  const Tensor<T> eps,
51  const std::vector<Function<T,NDIM> >& Vpsi1) const {
52 
53  double cpu0=cpu_time();
54  std::vector<Function<T,NDIM> > Vpsi=copy(world,Vpsi1);
55 
56  // add coupling between the functions (i.e. in case of localized orbitals)
57  // ( T + V ) \psi_i = \sum_j \psi_j F_{ji}
58  // ( T - F_{ii}) \psi_i = -V \psi_i + \sum_{j\neq i} \psi_jF_{ji}
59  // no coupling means F_{ij} =\eps_i \delta_{ij}, and the coupling term vanishes
60  Vpsi-=add_coupling_and_levelshift(psi,eps);
61 
62  std::vector < std::shared_ptr<SeparatedConvolution<double,NDIM> > > ops(psi.size());
63  for (int i=0; i<eps.dim(0); ++i) {
64  T e_i= (eps.ndim()==2) ? eps(i,i) : eps(i);
65  ops[i]=std::shared_ptr<SeparatedConvolution<double,NDIM> >(
66  BSHOperatorPtr<NDIM>(world, sqrt(-2.0*eps_in_green(e_i)), lo, bshtol));
67  ops[i]->destructive()=true;
68  }
69 
70  const std::vector<Function<T,NDIM> > tmp = apply(world,ops,-2.0*Vpsi);
71  const std::vector<Function<T,NDIM> > res=truncate(psi-tmp,FunctionDefaults<NDIM>::get_thresh()*0.1);
72  const std::vector<Function<T,NDIM> > bra_res=make_bra(res);
73 
74  // update eps
75  Tensor<double> norms=real(inner(world,make_bra(tmp),tmp));
76  Tensor<T> rnorms=inner(world,bra_res,res);
77  for (int i=0; i<norms.size(); ++i) {
78  norms(i)=sqrt(norms(i));
79  rnorms(i)=sqrt(rnorms(i));
80  }
81  if (printme) {
82  print("norm2(tmp)",norms);
83  print("norm2(res)",rnorms);
84  }
85 
86  Tensor<T> in=inner(world,Vpsi,bra_res); // no shift here!
87  Tensor<double> delta_eps(psi.size());
88  for (size_t i=0; i<psi.size(); ++i) delta_eps(i)=std::real(in(i))/(norms[i]*norms[i]);
89 
90  if (printme) print("orbital energy update",delta_eps);
91  double cpu1=cpu_time();
92  if (printme) printf("time in BSHApply() %8.4fs\n",cpu1-cpu0);
93 
94  return std::make_tuple(res,delta_eps);
95  }

123  std::vector<Function<T,NDIM> > add_coupling_and_levelshift(const std::vector<Function<T,NDIM> > psi,
124  const Tensor<T> fock1) const {
125 
126  // check dimensions
127  bool consistent=(psi.size()==size_t(fock1.dim(0)));
128  if ((fock1.ndim()==2) and not (psi.size()==size_t(fock1.dim(1)))) consistent=false;
129 
130  if (not consistent) {
131  print("Fock matrix dimensions",fock1.ndim(), fock1.dim(0), fock1.dim(1));
132  print("number of orbitals",psi.size());
133  MADNESS_EXCEPTION("confused Fock matrix/orbital energies in BSHApply - 1",1);
134  }
135 
136  bool do_coupling=(fock1.ndim()==2);
137 
138  // subtract the BSH energy (aka Fock diagonal elements) from the rhs
139  // ( T - fock(i,i) ) psi_i = -V psi_i + \sum_{j\neq i} psi_j fock(j,i)
140  // if there is no level shift and the orbital energies are large enough the
141  // diagonal should simply vanish.
142  if (do_coupling) {
143  Tensor<T> fock=copy(fock1);
144  for (int i=0; i<fock.dim(0); ++i) {
145  fock(i,i)-=eps_in_green(fock(i,i));
146  }
147  return transform(world, psi, fock);
148 
149  } else {
150  std::vector<T> eps(psi.size());
151  if (fock1.ndim()==1)
152  for (int i=0; i<fock1.dim(0); ++i) eps[i]=fock1(i)-eps_in_green(fock1(i));
153  if (fock1.ndim()==2)
154  for (int i=0; i<fock1.dim(0); ++i) eps[i]=fock1(i,i)-eps_in_green(fock1(i,i));
155 
156  std::vector<Function<T,NDIM> > result=copy(world,psi);
157  scale(world,result,eps);
158  return result;
159  }
