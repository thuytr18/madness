117 /// store and load data to/from the cloud into arbitrary worlds
118 ///
119 /// Distributed data is always bound to a certain world. If it needs to be
120 /// present in another world it can be serialized to the cloud and deserialized
121 /// from there again. For an example see test_cloud.cc
122 ///
123 /// Data is stored into a distributed container living in the universe.
124 /// During storing a (replicated) list of records is returned that can be used to find the data
125 /// in the container. If a combined object (a vector, tuple, etc) is stored a list of records
126 /// will be generated. When loading the data from the world the record list will be used to
127 /// deserialize all stored objects.
128 ///
129 /// Note that there must be a fence after the destruction of subworld containers, as in:
130 ///
131 /// create subworlds
132 /// {
133 /// dcT(subworld)
134 /// do work
135 /// }
136 /// subworld.gop.fence();

270  void replicate(const std::size_t chunk_size=INT_MAX) {
271 
272  World& world=container.get_world();
273  cloudtimer t(world,replication_time);
274  container.reset_pmap_to_local();
275  is_replicated=true;
276 
277  std::list<keyT> keylist;
278  for (auto it=container.begin(); it!=container.end(); ++it) {
279  keylist.push_back(it->first);
280  }
281 
282  for (ProcessID rank=0; rank<world.size(); rank++) {
283  if (rank == world.rank()) {
284  std::size_t keylistsize = keylist.size();
285  world.mpi.Bcast(&keylistsize,sizeof(keylistsize),MPI_BYTE,rank);
286 
287  for (auto key : keylist) {
288  madness::WorldContainer<keyT, std::vector<unsigned char> >::const_accessor acc;
289  bool found=container.find(acc,key);
290  MADNESS_CHECK(found);
291  auto data = acc->second;
292  std::size_t sz=data.size();
293 
294  world.mpi.Bcast(&key,sizeof(key),MPI_BYTE,rank);
295  world.mpi.Bcast(&sz,sizeof(sz),MPI_BYTE,rank);
296 
297  // if data is too large for MPI_INT break it into pieces to avoid integer overflow
298  for (std::size_t start=0; start<sz; start+=chunk_size) {
299  std::size_t remainder = std::min(sz - start, chunk_size);
300  world.mpi.Bcast(&data[start], remainder, MPI_BYTE, rank);
301  }
302 
303  }
304  }
305  else {
306  std::size_t keylistsize;
307  world.mpi.Bcast(&keylistsize,sizeof(keylistsize),MPI_BYTE,rank);
308  for (size_t i=0; i<keylistsize; i++) {
309  keyT key;
310  world.mpi.Bcast(&key,sizeof(key),MPI_BYTE,rank);
311  std::size_t sz;
312  world.mpi.Bcast(&sz,sizeof(sz),MPI_BYTE,rank);
313  valueT data(sz);
314 // world.mpi.Bcast(&data[0],sz,MPI_BYTE,rank);
315  for (std::size_t start=0; start<sz; start+=chunk_size) {
316  std::size_t remainder=std::min(sz-start,chunk_size);
317  world.mpi.Bcast(&data[start],remainder,MPI_BYTE,rank);
318  }
319 
320  container.replace(key,data);
321  }
322  }
323  }
324  world.gop.fence();
325  }
